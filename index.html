<script>

function checkAnswer(inputId, correctKeywords, nextCardId) {

    const val = document.getElementById(inputId).value.trim();

    const feedback = document.getElementById(inputId + "-result");

 

    const ok = correctKeywords.some(k => val.includes(k));

 

    if (ok) {

        feedback.style.color = "#7CFC00";

        feedback.innerHTML = "✔ 正確！前往下一關！";

 

        document.getElementById(nextCardId).classList.add("active");

        document.getElementById(nextCardId).scrollIntoView({ behavior: "smooth" });

 

    } else {

        feedback.style.color = "#ff6b6b";

        feedback.innerHTML = "✘ 再想想看！";

    }

}
